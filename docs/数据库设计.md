## 2. SQLite DDL建表语句

```sql
CREATE TABLE IF NOT EXISTS users (
    id INTEGER PRIMARY KEY AUTOINCREMENT COMMENT '用户ID（主键，自增）',
    phone VARCHAR(20) NOT NULL UNIQUE COMMENT '手机号（唯一索引）',
    nickname VARCHAR(100) DEFAULT '' COMMENT '用户昵称',
    avatar_url VARCHAR(500) DEFAULT '' COMMENT '头像URL',
    password_hash VARCHAR(255) DEFAULT '' COMMENT '密码哈希',
    wx_openid VARCHAR(100) UNIQUE COMMENT '微信OpenID',
    status INTEGER DEFAULT 1 COMMENT '用户状态（0禁用，1启用）',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP COMMENT '更新时间'
);

CREATE TABLE IF NOT EXISTS devices (
    id INTEGER PRIMARY KEY AUTOINCREMENT COMMENT '设备ID（主键，自增）',
    user_id INTEGER NOT NULL COMMENT '所属用户ID',
    device_name VARCHAR(100) NOT NULL COMMENT '设备名称',
    device_type VARCHAR(50) COMMENT '设备类型（如：智能床垫、空气净化器）',
    device_sn VARCHAR(50) UNIQUE COMMENT '设备序列号',
    status VARCHAR(20) DEFAULT 'offline' COMMENT '设备状态（online/offline）',
    temperature REAL DEFAULT 0 COMMENT '当前温度',
    humidity REAL DEFAULT 0 COMMENT '当前湿度',
    noise_level REAL DEFAULT 0 COMMENT '当前噪音值',
    last_sync_time DATETIME COMMENT '最后同步时间',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP COMMENT '更新时间',
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE
);

CREATE TABLE IF NOT EXISTS sleep_reports (
    id INTEGER PRIMARY KEY AUTOINCREMENT COMMENT '报告ID（主键，自增）',
    user_id INTEGER NOT NULL COMMENT '所属用户ID',
    report_date DATE NOT NULL COMMENT '报告日期',
    total_sleep_minutes INTEGER DEFAULT 0 COMMENT '总睡眠时长（分钟）',
    deep_sleep_minutes INTEGER DEFAULT 0 COMMENT '深睡眠时长（分钟）',
    light_sleep_minutes INTEGER DEFAULT 0 COMMENT '浅睡眠时长（分钟）',
    rem_sleep_minutes INTEGER DEFAULT 0 COMMENT 'REM睡眠时长（分钟）',
    sleep_score INTEGER DEFAULT 0 COMMENT '睡眠评分（0-100）',
    sleep_stages TEXT COMMENT '睡眠分期数据（JSON格式）',
    note TEXT COMMENT '备注说明',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE,
    UNIQUE(user_id, report_date)
);

CREATE TABLE IF NOT EXISTS user_settings (
    id INTEGER PRIMARY KEY AUTOINCREMENT COMMENT '设置ID（主键，自增）',
    user_id INTEGER NOT NULL UNIQUE COMMENT '所属用户ID（唯一）',
    wake_up_time TIME COMMENT '起床时间',
    sleep_time TIME COMMENT '就寝时间',
    noise_threshold INTEGER DEFAULT 50 COMMENT '噪音阈值（分贝）',
    temp_min INTEGER DEFAULT 18 COMMENT '温度下限（℃）',
    temp_max INTEGER DEFAULT 28 COMMENT '温度上限（℃）',
    humidity_min INTEGER DEFAULT 30 COMMENT '湿度下限（%）',
    humidity_max INTEGER DEFAULT 70 COMMENT '湿度上限（%）',
    notification_enabled INTEGER DEFAULT 1 COMMENT '是否开启通知（0关闭，1开启）',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP COMMENT '更新时间',
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE
);

CREATE TABLE IF NOT EXISTS doctor_authorizations (
    id INTEGER PRIMARY KEY AUTOINCREMENT COMMENT '授权ID（主键，自增）',
    user_id INTEGER NOT NULL COMMENT '授权用户ID',
    doctor_name VARCHAR(100) NOT NULL COMMENT '医生姓名',
    doctor_id VARCHAR(50) COMMENT '医生工号',
    hospital VARCHAR(200) COMMENT '所属医院',
    department VARCHAR(100) COMMENT '所属科室',
    status INTEGER DEFAULT 0 COMMENT '授权状态（0待审核，1已授权，2已撤销）',
    authorized_at DATETIME COMMENT '授权时间',
    revoked_at DATETIME COMMENT '撤销时间',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP COMMENT '更新时间',
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE
);

CREATE INDEX idx_devices_user_id ON devices(user_id);
CREATE INDEX idx_sleep_reports_user_id ON sleep_reports(user_id);
CREATE INDEX idx_sleep_reports_report_date ON sleep_reports(report_date);
CREATE INDEX idx_doctor_authorizations_user_id ON doctor_authorizations(user_id);
CREATE INDEX idx_doctor_authorizations_status ON doctor_authorizations(status);
```
